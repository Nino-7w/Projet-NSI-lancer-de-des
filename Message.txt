import random
import tkinter as tk

# ----------------------------
# Données
# ----------------------------
dés = {"Dé rouge" : 6, "Dé surprise" : random.randint(1,100) , "Dé Casino" : 777 ,
       "Dé blanc" : 6, "Dé ténébres" : 16, "Dé Chromatique" : 150 , "Dé Archange" : 4000 , 
       "Dé carré" : 4, "Dé lettres" : 9 , "Dé Démon" : 666, "Dé Infini" : 500000 , 
       "Dé pyramide" : 20, "Dé boudchou" : 40 , "Dé Ange" : 1000  , "Dé Monstrueux" : 500 , 
       "Dé Géant" : 60 , "Dé Vitré" : 45 , "Dé Gargantua" : 2000 , "Dé Vieux" : 3 , 
       } 

# ----------------------------
# Fonctions
# ----------------------------
def lancer_de_dés():
    resultat = []
    details = []
    nb_dés = 5

    for _ in range(nb_dés):
        nom, faces = random.choice(list(dés.items()))
        valeur = random.randint(1, faces)
        resultat.append(valeur)
        details.append(f"{nom} ({faces}) → {valeur}")

    return resultat, details


def lancer():
    zone_resultat.delete("1.0", tk.END)

    valeurs, details = lancer_de_dés()
    total = sum(valeurs)

    for ligne in details:
        zone_resultat.insert(tk.END, ligne + "\n")

    label_score.config(text=f"Score total : {total}")

# ----------------------------
# Interface graphique
# ----------------------------
fenetre = tk.Tk()
fenetre.title("Projet : Lancer de dés 🎲")
fenetre.geometry("400x300")

titre = tk.Label(fenetre, text="Lancer de dés", font=("Arial", 16, "bold"))
titre.pack(pady=10)

bouton = tk.Button(fenetre, text="Lancer les dés", command=lancer)
bouton.pack(pady=5)

zone_resultat = tk.Text(fenetre, height=8, width=40)
zone_resultat.pack(pady=10)

label_score = tk.Label(fenetre, text="Score total : 0", font=("Arial", 12))
label_score.pack()

fenetre.mainloop()
